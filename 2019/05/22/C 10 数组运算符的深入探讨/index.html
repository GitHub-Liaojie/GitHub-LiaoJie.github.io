数组运算符的深入探讨

在前面已经知道数组可以随机初始化了。
那么此时可以对数组做一定运算和处理
得到需要的结果。
eg：
//数组求和，length代表数组长度
float arr_get_total(int *p,int length)
{
int i = 0;
int sum = 0;
for (i = 0; i < length; i++)
{
sum = sum + p[i];
}
return sum;
}
//数组求平均，length代表数组长度
double arr_get_average(int *p,int length)
{
int i = 0;
int sum = 0;
for (i = 0; i < length; i++)
{
sum = sum + p[i];
}
return (double)sum/length;
}
//数组相乘，length代表数组长度
float arr_get_multi(int *p, int length)
{
int i = 0;
int sum = 1;
for (i = 0; i < length; i++)
{
sum = sum * p[i];
}
return sum ;
}
float arr_my_math(int *p,int length, double(*p1)(int *p1, int length1))
{
float a = 0;
a = (float)p1(p , length);
return a;
}
通过调用以上函数则可以对数组进行求和/乘法运算/求平均
等等运算。这样均可以得到需要的结果。
如：
//用下面的方式 直接调用函数即可直接运行
int main()
{
avg = arr_my_math(a,length, arr_get_average);
printf(“一维数组平均数为%lf\n”, avg);
avg = arr_my_math(a, length, arr_get_multi);
printf(“一维数组相乘为%lf\n”, avg);
}
